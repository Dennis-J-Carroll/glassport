```python
# src/glassport/detectors.py — replace _normalize_for_scan (:587) with:
def _normalize_with_map(text: str) -> tuple[str, list[int]]:
    """Normalize like _normalize_for_scan, but char-by-char, recording for
    each emitted normalized char the index in the ORIGINAL string it came
    from. invisible → dropped (no entry); confusable → 1:1; NFKC per char →
    0..N chars, all tagged to that one origin index. The map lets a match
    found in normalized space be redacted back in the original bytes.

    Per-char NFKC differs from whole-string NFKC only for cross-char
    combining sequences (irrelevant to ASCII-ish credentials); the
    fail-closed backstop in redact_secrets_strict proves any drift is safe."""
    out: list[str] = []
    origin: list[int] = []
    for i, ch in enumerate(text):
        if _INVISIBLE_RE.match(ch):
            continue
        for nc in unicodedata.normalize("NFKC", ch.translate(_CONFUSABLES)):
            out.append(nc)
            origin.append(i)
    return "".join(out), origin


def _normalize_for_scan(text: str) -> str:
    """Defeat obfuscation before pattern-matching. See _normalize_with_map."""
    return _normalize_with_map(text)[0]
```

```python
def _scan_normalized(text: str) -> list[tuple[PIIPattern, str, int, int]]:
    """Validated, de-duped PII hits WITH normalized-coordinate spans.
    `text` is assumed already normalized (see _normalize_for_scan)."""
    raw: list[tuple[PIIPattern, str, int, int]] = []
    structural_spans: list[tuple[int, int]] = []
    for pat in _active_patterns():
        for m in pat.pattern.finditer(text):
            value = m.group(m.lastindex) if m.lastindex else m.group(0)
            if pat.validator and not pat.validator(value):
                continue
            span = m.span(m.lastindex) if m.lastindex else m.span(0)
            raw.append((pat, value, span[0], span[1]))
            if pat.category in _STRUCTURAL_CONTAINERS:
                structural_spans.append(span)

    hits: list[tuple[PIIPattern, str, int, int]] = []
    seen: set[tuple[str, str]] = set()
    for pat, value, start, end in raw:
        if pat.category in _GENERIC_SECRETS and any(
                s <= start and end <= e for s, e in structural_spans):
            continue
        dedup = (pat.category, value)
        if dedup in seen:
            continue
        seen.add(dedup)
        hits.append((pat, value, start, end))
    return hits


def _scan_pii_spanned(text: str) -> list[tuple[PIIPattern, str, int, int]]:
    if len(text) > MAX_SCAN_BYTES:
        text = text[:MAX_SCAN_BYTES]
    return _scan_normalized(_normalize_for_scan(text))


def _scan_pii(text: str) -> list[tuple[PIIPattern, str]]:
    """Validated, de-duplicated PII hits (no spans). Back-compat wrapper."""
    return [(p, v) for p, v, _, _ in _scan_pii_spanned(text)]
```

```python
# src/glassport/detectors.py — replace _apply_redactions (:703) and both
# redact_* functions with:

def _apply_span_redactions(text: str, spans: list[tuple[int, int, str, str]]) -> str:
    """Redact original-text ranges. spans = (start, end, category, value),
    applied right-to-left so earlier offsets stay valid as the string shrinks."""
    for start, end, category, value in sorted(spans, reverse=True):
        text = text[:start] + _redact(value, category) + text[end:]
    return text


def _spanned_original_redactions(text: str) -> list[tuple[int, int, str, str]]:
    """Map every normalized-space hit back to an ORIGINAL-text range."""
    norm, origin = _normalize_with_map(text[:MAX_SCAN_BYTES])
    spans: list[tuple[int, int, str, str]] = []
    for pat, value, a, b in _scan_normalized(norm):
        if not value or b <= a:
            continue
        start = origin[a]
        end = origin[b - 1] + 1        # cover the whole source char (safe over-approx)
        spans.append((start, end, pat.category, value))
    return spans


def redact_secrets(text: str) -> str:
    """Best-effort scrubber (internal/analytical paths only). Returns the
    input UNCHANGED if the scan raises. For shareable artifacts use
    redact_secrets_strict()."""
    try:
        spans = _spanned_original_redactions(text)
    except Exception:
        return text
    return _apply_span_redactions(text, spans)


def redact_secrets_strict(text: str) -> str:
    """Fail-closed scrubber for shareable output (report HTML, SARIF).
    Withholds the whole field if (a) the scan raises, or (b) any secret
    survives redaction — a span-map imprecision must never leak a live key
    into an artifact that leaves the machine."""
    try:
        spans = _spanned_original_redactions(text)
        out = _apply_span_redactions(text, spans)
        if _scan_pii(out):              # backstop: re-scan the OUTPUT
            return _WITHHELD
    except Exception:
        return _WITHHELD
    return out
```

```python
# mcp_http.py — after _HOP (:30):
_HOP_TOKEN_RE = re.compile(r"^[!#$%&'*+.^_`|~0-9a-z-]+$")  # RFC 7230 token, lowercased

def _hop_headers(pairs) -> set[str]:
    """Static hop-by-hop set plus any field a Connection header nominates.
    A hostile upstream/client can list an extra header in Connection to make
    it connection-local; those must not be forwarded end-to-end."""
    drop = set(_HOP)
    for k, v in pairs:
        if k.lower() == "connection":
            for tok in v.split(","):
                name = tok.strip().lower()
                if name and _HOP_TOKEN_RE.match(name):
                    drop.add(name)
    return drop
```

```python
# mcp_http.py — near the top helpers:
def _validate_remote(url: str):
    r = urlsplit(url)
    if r.scheme not in ("http", "https"):
        raise ValueError(f"remote scheme {r.scheme!r} unsupported: use http/https")
    if not r.hostname:
        raise ValueError("remote URL has no host")
    if r.username or r.password:
        raise ValueError("remote URL must not embed credentials (userinfo)")
    if r.fragment:
        raise ValueError("remote URL must not contain a fragment")
    try:
        r.port                       # property raises ValueError if out of range
    except ValueError:
        raise ValueError("remote URL port is out of range")
    return r


def _host_header(remote) -> str:
    """Host from hostname (+ explicit non-default port), never raw netloc —
    netloc can carry userinfo the proxy must not forward."""
    host = remote.hostname or ""
    if ":" in host:                  # IPv6 literal
        host = f"[{host}]"
    default = 443 if remote.scheme == "https" else 80
    if remote.port and remote.port != default:
        return f"{host}:{remote.port}"
    return host
```

```python
# tap.py — inside SessionLog, after __init__:
def file_mode(self) -> int | None:
    """POSIX permission bits of the open log, or None (non-POSIX / error)."""
    if os.name != "posix":
        return None
    try:
        return os.fstat(self._fh.fileno()).st_mode & 0o777
    except OSError:
        return None

def close(self) -> None:
    try:
        self._fh.close()
    except Exception:
        pass
```

```python
# tap.py — module level, after SessionLog:
def open_session_log(path: Path) -> "SessionLog | None":
    """Create a verified-private session log, or return None so the relay
    runs without recording. The relay is sacred: neither an unwritable dir
    nor a permission we could not secure may kill the session — we disable
    logging loudly instead."""
    try:
        log = SessionLog(path)
    except OSError as exc:
        print(f"[glassport] logging disabled: cannot write {path.parent} "
              f"({exc}) -- relay continues", file=sys.stderr)
        return None
    mode = log.file_mode()
    if mode is not None and (mode & 0o077):
        print(f"[glassport] logging disabled: {path} is not owner-only "
              f"(mode {mode:04o}) -- refusing to record sensitive traffic to "
              f"a readable file; relay continues", file=sys.stderr)
        log.close()
        return None
    return log
```
